  SUM(o.amount) AS total_amount

ORDER BY total_amount DESC;

⚡ Задача 4: Async API + Exponential Backoff

Напишите функцию на aiohttp, которая делает запрос к API с максимум 5 попытками и экспоненциальной задержкой при ошибках 429/500.

✅ Показать эталон 
import asyncio
import aiohttp

async def fetch_with_retry(url, attempts=5):
    async with aiohttp.ClientSession() as session:
        for i in range(attempts):
            try:
                async with session.get(url, timeout=10) as resp:
                    if resp.status in (429, 500, 502, 503, 504):
                        raise RuntimeError(f"retryable status={resp.status}")
                    resp.raise_for_status()
                    return await resp.json()
            except Exception:
                if i == attempts - 1:
                    raise
                await asyncio.sleep(2 ** i)

Цель: Написать Kafka Producer на Python, который симулирует поток транзакций. Перед отправкой скрипт валидирует данные (DQ check): сумма (amount) должна быть > 0. Если данные невалидны — они отбрасываются.

События на генерацию:

Топик: "transactions"
✅ Показать эталон 
import json
from kafka import KafkaProducer

producer = KafkaProducer(
    bootstrap_servers=['localhost:9092'],
    value_serializer=lambda v: json.dumps(v).encode('utf-8')
)

transactions = [
    {'txn_id': 1, 'amount': 150},
    {'txn_id': 2, 'amount': -10},
    {'txn_id': 3, 'amount': 200}
]

for txn in transactions:
    # DQ Check
    if txn['amount'] <= 0:
        print(f"DQ Alert: Invalid amount for txn_id {txn['txn_id']}")
        continue

    # Send to Kafka
    producer.send('transactions', value=txn)

producer.flush()
print("Valid transactions sent.")

💻 Проект 1: Поиск математических аномалий (DQ Check)

Цель: Написать Python функцию detect_outliers(data), которая принимает список чисел и возвращает список подозрительных значений. Будем использовать межквартильный размах (IQR): аномалия — число, находящееся за пределами `Q1 - 1.5 * IQR` и `Q3 + 1.5 * IQR`.

Пример вызова:
data = [10, 12, 12, 13, 12, 11, 14, 13, 15, 102, 12, 14]
# В этом массиве число 102 явно отклоняется от среднего.
detect_outliers(data) -> [102]
✅ Показать эталон 
import numpy as np

def detect_outliers(data):
    # 1. Считаем 1-й и 3-й квартили (25й и 75й процентили)
    q1, q3 = np.percentile(data, [25, 75])
    
    # 2. Вычисляем IQR (межквартильный размах)
    iqr = q3 - q1
    
    # 3. Высчитываем границы (lower_bound и upper_bound)
    lower_bound = q1 - (1.5 * iqr)
    upper_bound = q3 + (1.5 * iqr)
    
    # 4. Собираем список выбросов
    outliers = [x for x in data if x < lower_bound or x > upper_bound]
    
    return outliers

# Проверка:
print(detect_outliers([10, 12, 12, 13, 12, 11, 14, 13, 15, 102, 12, 14]))
# Вывод: [102]

from math import sqrt

def estimate_sample_size(p, delta, z_alpha=1.96, z_beta=0.84):
    p2 = p + delta
    var = p * (1 - p) + p2 * (1 - p2)
    n = ((z_alpha + z_beta) ** 2 * var) / (delta ** 2)
    return int(n) + 1

📉 Проект 3: Z-score мониторинг метрики

Есть дневная метрика revenue. Напишите логику аномалий: если |z| > 3, поднимаем alert. Можно Python или SQL-псевдокод.

✅ Показать эталон 
import statistics

def detect_zscore_anomalies(series, threshold=3):
    mu = statistics.mean(series)
    sigma = statistics.pstdev(series)
    if sigma == 0:
        return []
    return [x for x in series if abs((x - mu) / sigma) > threshold]
